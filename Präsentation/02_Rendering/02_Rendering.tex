\begin{frame}{\Huge{Fazit}}
	
		\begin{block}{Vorteile}
			\begin{itemize}
				\item Geometrie- und Materialeigenschaften für lokale Szene müssen nur einmal bestimmt werden
				\item Lichtbasiertes Model der Distanzszene muss nur einmal bestimmt werden
				\item Beleuchtung wirkt korrekt auf synthetische Objekte ein
			\end{itemize}
		\end{block}
		
		
		\begin{block}{Nachteile}
			\begin{itemize}
				\item BRDFs für lokale Szene können nur geschätzt werden
				\item Kamera muss kalibriert werden
			\end{itemize}
		\end{block}
\end{frame}
